function shellSort(nums) {
  const len = nums.length;
  for (let gap = Math.floor(len / 2); gap >= 1; gap = Math.floor(gap / 2)) {
    for (let i = gap; i < len; i += gap) {
      for (let j = i; j > 0 && nums[j] < nums[j - gap]; j -= gap) {
        const temp = nums[j];
        nums[j] = nums[j - gap];
        nums[j - gap] = temp;
      }
    }
  }
  return nums;
}

        const temp = nums[j];
